Cumulative table values

1. Let's create a tables:

CREATE TABLE CUMULATIVESUM (
  ID tinyint
  ,VALUE tinyint
) ENGINE=INNODB

2. And populate it with some data:

INSERT INTO CUMULATIVESUM
SELECT 1, 10 UNION
SELECT 2, 2  UNION
SELECT 3, 6  UNION
SELECT 4, 10

 

3. Verify data:

/* Department table */
SELECT
    *
FROM
    CUMULATIVESUM


/*
Listing table values
*/

4. Self-join the table:

SELECT
    t1.ID
    ,t1.VALUE
    ,t2.VALUE
FROM
    CUMULATIVESUM t1
JOIN CUMULATIVESUM t2 ON t1.ID >= t2.ID
ORDER By t1.ID

4. Cumulative self-join:

SELECT
    t1.ID
    ,t1.VALUE
    ,Sum(t2.VALUE) as CUMULATIVE_VALUE
FROM
    CUMULATIVESUM t1
    ,CUMULATIVESUM t2
WHERE t1.ID >= t2.ID
GROUP BY t1.ID, t1.VALUE
ORDER BY t1.ID
